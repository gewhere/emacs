  "Show a popup if we're on X, or echo it otherwise; TITLE is the title

a sound to be played"

  (interactive)
  (when sound (shell-command

      (shell-command (concat "notify-send "

                             (if icon (concat "-i " icon) "")
                             " '" title "' '" msg "'"))
    ;; text only version

    (message (concat title ": " msg))))
